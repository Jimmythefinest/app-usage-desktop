```dataviewjs const ACTIVITIES_PATH = "Activities"; // folder containing Activity notes const typeName = "Activity"; const activities = dv.pages(`"${ACTIVITIES_PATH}"`) .where(p => p.type === typeName && p.start && p.elapsed_seconds != null); // Extract unique dates (YYYY-MM-DD) from the `start` field const dates = Array.from(new Set( activities .map(p => dv.date(p.start).toFormat("yyyy-MM-dd")) .values() )).sort(); const container = dv.container; container.innerHTML = ""; const todayStr = dv.date("today").toFormat("yyyy-MM-dd"); const defaultDate = dates.includes(todayStr) ? todayStr : (dates[0] ?? App usage by day
`); const selectEl = wrapper.querySelector("#usageByDaySelect"); const bodyEl = wrapper.querySelector("#usageByDayBody"); for (const d of dates) { const opt = document.createElement("option"); opt.value = d; opt.textContent = d; if (d === defaultDate) opt.selected = true; selectEl.appendChild(opt); } function fmtSeconds(sec) { sec = Number(sec || 0); const h = Math.floor(sec / 3600); const m = Math.floor((sec % 3600) / 60); const s = Math.floor(sec % 60); if (h > 0) return `${h}h ${m}m`; if (m > 0) return `${m}m ${s}s`; return `${s}s`; } function render(dateStr) { bodyEl.innerHTML = ""; const dayActs = activities .where(p => dv.date(p.start).toFormat("yyyy-MM-dd") === dateStr); // Aggregate focused time by app const byApp = new Map(); for (const a of dayActs.values) { const app = a.app ?? "(unknown)"; const elapsed = Number(a.elapsed_seconds ?? 0); const prev = byApp.get(app) ?? 0; byApp.set(app, prev + elapsed); } const rows = Array.from(byApp.entries()) .map(([app, seconds]) => ({ app, seconds })) .sort((x, y) => y.seconds - x.seconds); const totalSeconds = rows.reduce((sum, r) => sum + r.seconds, 0); // Header bodyEl.insertAdjacentHTML("beforeend", `

`); // Table const table = document.createElement("table"); table.className = "usage-by-day-table"; table.innerHTML = ` App Focused Share `; const tbody = table.querySelector("tbody"); for (const r of rows) { const share = totalSeconds > 0 ? (r.seconds / totalSeconds) : 0; const tr = document.createElement("tr"); tr.innerHTML = ` ${r.app} ${fmtSeconds(r.seconds)}
${Math.round(share * 100)}%
`; tbody.appendChild(tr); } bodyEl.appendChild(table); } selectEl.addEventListener("change", (e) => render(e.target.value)); render(defaultDate); ```
